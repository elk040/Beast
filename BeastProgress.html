def compute_ssrt(df, stop_go_col, stop_acc_col, ssd_col, rt_col, go_code, stop_code):
    if rt_col not in df.columns:
        raise ValueError(f"RT column '{rt_col}' not found in dataframe!")

    # Map stop/go to numeric 1 = stop, 2 = go
    df = df.copy()
    df['stop_trial'] = df[stop_go_col].map({stop_code: 1, go_code: 2})

    go_trials = df[df['stop_trial'] == 2]
    stop_trials = df[df['stop_trial'] == 1]

    if len(go_trials) == 0 or len(stop_trials) == 0:
        return np.nan

    p_respond = 1 - stop_trials[stop_acc_col].mean()
    go_rts = np.sort(go_trials[rt_col].values)
    n_idx = int(np.ceil(p_respond * len(go_rts))) - 1
    n_idx = max(n_idx, 0)
    nth_rt = go_rts[n_idx]
    mean_ssd = stop_trials[ssd_col].mean()

    return nth_rt - mean_ssd

import os
import pandas as pd
import numpy as np
import nibabel as nib
from nilearn.glm.first_level import FirstLevelModel
from nilearn import plotting

# ------------------------------
# Dataset paths
# ------------------------------
dataset_path = "/Users/elizabethkudriavtsev/fMRI_data/ds002316"
sub = "sub-119"
run = "run-1"

bold_img_path = os.path.join(dataset_path, sub, "func",
                             f"{sub}_task-Beast_run-1_bold.nii.gz")
events_file_path = os.path.join(dataset_path, sub, "func",
                                f"{sub}_task-Beast_run-1_events.tsv")

# ------------------------------
# Check files exist
# ------------------------------
print("BOLD exists:", os.path.exists(bold_img_path))
print("Events exists:", os.path.exists(events_file_path))

# ------------------------------
# Load BOLD to inspect
# ------------------------------
img = nib.load(bold_img_path)
print("BOLD shape:", img.shape)
n_volumes = img.shape[3]
print("Number of volumes:", n_volumes)

# Optional: take only first N volumes to reduce memory usage
max_vols = 100  # adjust if needed
if n_volumes > max_vols:
    print(f"Truncating to first {max_vols} volumes for testing...")
    img = nib.Nifti1Image(img.get_fdata()[..., :max_vols], img.affine, img.header)

# ------------------------------
# Load and preprocess events
# ------------------------------
events = pd.read_csv(events_file_path, sep="\t")

# Keep only Go/Stop trials
events = events[events['new_trial_type'].isin(['go_stim', 'stop_stim'])].copy()

# Convert numeric columns safely
numeric_cols = ['onset', 'duration', 'rt', 'go_acc', 'stop_acc', 'SSD_target']
for col in numeric_cols:
    events[col] = pd.to_numeric(events[col], errors='coerce')

# Drop rows with missing onset/duration
events = events.dropna(subset=['onset', 'duration']).reset_index(drop=True)

# Minimal regressor
events['go_trial'] = (events['new_trial_type'] == 'go_stim').astype(int)
design_df = events[['onset', 'duration', 'go_trial']].copy()

print("Design matrix preview:")
print(design_df.head())

# ------------------------------
# Fit first-level GLM
# ------------------------------
tr = 0.68  # seconds
first_level_model = FirstLevelModel(
    t_r=tr,
    slice_time_ref=0.5,
    smoothing_fwhm=5.0,
    noise_model='ar1',
    standardize=False
)

first_level_model = first_level_model.fit(img, events=design_df)
print("GLM fitted successfully!")

# ------------------------------
# Compute contrast
# ------------------------------
contrast_vec = np.array([1])  # Go > Baseline
z_map = first_level_model.compute_contrast(contrast_vec, output_type='z_score')

# ------------------------------
# Plot results
# ------------------------------
plotting.plot_stat_map(z_map, title='Go > Baseline', display_mode='ortho', threshold=3.0)
